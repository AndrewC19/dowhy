data["df"].rename(columns={'y':'behavior'}, inplace=True)
model = CausalModel(
    data=data['df'],
    treatment=data["treatment_name"],
    outcome="behavior",
    graph=data["dot_graph"].replace("y", "behavior"),
    proceed_when_unidentifiable=True,
    test_significance=None
)
target_estimand = model.identify_effect(method_name="auto")
print(target_estimand)
ate_estimate = model.estimate_effect(
    identified_estimand=target_estimand,
    method_name="frontdoor.two_stage_regression",
    test_significance=None,
    method_params = {
                                            'first_stage_model': dowhy.causal_estimators.linear_regression_estimator.LinearRegressionEstimator,
                                            'second_stage_model': dowhy.causal_estimators.linear_regression_estimator.LinearRegressionEstimator
                                        }
)
print(ate_estimate)
true_ate = data["ate"]
print(true_ate, ate_estimate.value)
print(target_estimand)

def simple_linear_outcome_model(X_train, output_train):
    # The outcome is a linear function of the confounder
    # The slope is 1,2 and the intercept is 3
    return lambda X_train:  X_train[:,0] + 2*X_train[:,1] + 3
